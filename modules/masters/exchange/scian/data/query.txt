-- CREATE TABLES

CREATE TABLE sector
(
  sector_id character varying(10) NOT NULL,
  sector_title character varying(255) NOT NULL,
  CONSTRAINT sector_pkey PRIMARY KEY (sector_id),
  CONSTRAINT sector_sector_title_key UNIQUE (sector_title)
)
WITH (
  OIDS=FALSE
);

CREATE TABLE subsector
(
  subsector_id character varying(10) NOT NULL,
  sector_id character varying(10) NOT NULL,
  subsector_title character varying(255) NOT NULL,
  CONSTRAINT subsector_pkey PRIMARY KEY (subsector_id),
  CONSTRAINT subsector_sector_id_fkey FOREIGN KEY (sector_id)
      REFERENCES sector (sector_id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT subsector_subsector_title_key UNIQUE (subsector_title)
)
WITH (
  OIDS=FALSE
);

CREATE TABLE rama
(
  rama_id character varying(10) NOT NULL,
  subsector_id character varying(10) NOT NULL,
  rama_title character varying(255) NOT NULL,
  CONSTRAINT rama_pkey PRIMARY KEY (rama_id),
  CONSTRAINT rama_subsector_id_fkey FOREIGN KEY (subsector_id)
      REFERENCES subsector (subsector_id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT rama_rama_title_key UNIQUE (rama_title)
)
WITH (
  OIDS=FALSE
);

CREATE TABLE subrama
(
  subrama_id character varying(10) NOT NULL,
  rama_id character varying(10) NOT NULL,
  subrama_title character varying(255) NOT NULL,
  CONSTRAINT subrama_pkey PRIMARY KEY (subrama_id),
  CONSTRAINT subrama_rama_id_fkey FOREIGN KEY (rama_id)
      REFERENCES rama (rama_id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT subrama_subrama_title_key UNIQUE (subrama_title)
)
WITH (
  OIDS=FALSE
);

CREATE TABLE clase
(
  clase_id character varying(10) NOT NULL,
  subrama_id character varying(10) NOT NULL,
  clase_title character varying(255) NOT NULL,
  CONSTRAINT clase_pkey PRIMARY KEY (clase_id),
  CONSTRAINT clase_subrama_id_fkey FOREIGN KEY (subrama_id)
      REFERENCES subrama (subrama_id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT clase_clase_title_key UNIQUE (clase_title)
)
WITH (
  OIDS=FALSE
);
